# Minimi CMake-version
cmake_minimum_required(VERSION 3.13)

# Projektets namn
project(PicoBackingtracksTransport)

# Ange C++-standarden
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Inkludera Raspberry Pi Pico SDK
include(pico_sdk_import.cmake)

# Skapa projektet
pico_sdk_init()

# Lägg till exekverbara filer
add_executable(PicoBackingtracksTransport main.cpp)

# Inkludera TinyUSB och Pico SDK-bibliotek
target_sources(PicoBackingtracksTransport PRIVATE main.cpp)
target_link_libraries(PicoBackingtracksTransport pico_stdlib pico_unique_id tinyusb_device tinyusb_board)

# Aktivera USB och stdio
pico_enable_stdio_usb(PicoBackingtracksTransport 1)
pico_enable_stdio_uart(PicoBackingtracksTransport 0)

# Lägg till blinkexempel för att kompilera till Pico
pico_add_extra_outputs(PicoBackingtracksTransport)
